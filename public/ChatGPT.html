To convert an object of objects into an array of objects, you can use the Object.values() method if you only want the values, or Object.entries() to also include the keys.

Here’s an example:

Example Input

javascript
const data = {
  obj1: { id: 1, name: 'Alice' },
  obj2: { id: 2, name: 'Bob' },
  obj3: { id: 3, name: 'Charlie' }
};

Using Object.values() (Only Values)

javascript
const arrayOfObjects = Object.values(data);

console.log(arrayOfObjects);
// Output: [
//   { id: 1, name: 'Alice' },
//   { id: 2, name: 'Bob' },
//   { id: 3, name: 'Charlie' }
// ]

Using Object.entries() (Including Keys)

If you also need the keys as part of the objects:

javascript
const arrayWithKeys = Object.entries(data).map(([key, value]) => {
  return { key, ...value };
});

console.log(arrayWithKeys);
// Output: [
//   { key: 'obj1', id: 1, name: 'Alice' },
//   { key: 'obj2', id: 2, name: 'Bob' },
//   { key: 'obj3', id: 3, name: 'Charlie' }
// ]

Choose the method that best fits your needs!

            `1 ${initialCrypto.name} = ${conversion.toFixed(6)} ${initialFiat.name}`
          );
        }
      })
      .catch((err) => console.error("Error fetching rates:", err));
  }, []);

  const handleConvert = (exchangeState = exchange) => {
    if (!rate1 || !rate2 || !amount || isNaN(amount)) {
      setResult("Please select valid rates and enter a positive amount.");
      return;
    }

    const fromRate = exchangeState ? rate2 : rate1;
    const toRate = exchangeState ? rate1 : rate2;

    const conversion = (parseFloat(amount) * fromRate.value) / toRate.value;

    setResult(
      `${amount} ${fromRate.name} = ${conversion.toFixed(6)} ${toRate.name}`
    );
  };

  const handleSwitch = () => {
    setExchange((prev) => {
      const updatedExchange = !prev;
      handleConvert(updatedExchange);
      return updatedExchange;
    });
  };

  const segregatedRates = {
    crypto: rates.filter((rate) => rate.type === "crypto"),
    fiat: rates.filter((rate) => rate.type === "fiat"),
  };

  useEffect(() => {
    if (rate1 && rate2 && amount) {
      handleConvert();
    }
  }, [amount, rate1, rate2]);

  return (
    <div className={s.convertorsection}>
      <h2>Crypto Converter Calculator</h2>
      <div className={s.inputsection}>
        <div>
          <input
            type="number"
            placeholder="Enter amount"
            value={amount}
            onChange={(e) => setAmount(e.target.value)}
          />
        </div>
        <div>
          <select
            value={rate1 ? JSON.stringify(rate1) : ""}
            onChange={(e) => setRate1(JSON.parse(e.target.value))}
          >
            <optgroup label="Crypto">
              {segregatedRates.crypto.map((rate, index) => (
                <option key={index} value={JSON.stringify(rate)}>
                  {rate.name} ({rate.unit})
                </option>
              ))}
            </optgroup>
            <optgroup label="Fiat">
              {segregatedRates.fiat.map((rate, index) => (
                <option key={index} value={JSON.stringify(rate)}>
                  {rate.name} ({rate.unit})
                </option>
              ))}
            </optgroup>
          </select>
          <button onClick={handleSwitch}>Switch</button>
          <select
            value={rate2 ? JSON.stringify(rate2) : ""}
            onChange={(e) => setRate2(JSON.parse(e.target.value))}
          >
            <optgroup label="Fiat">
              {segregatedRates.fiat.map((rate, index) => (
                <option key={index} value={JSON.stringify(rate)}>
                  {rate.name} ({rate.unit})
                </option>
              ))}
            </optgroup>
            <optgroup label="Crypto">
              {segregatedRates.crypto.map((rate, index) => (
                <option key={index} value={JSON.stringify(rate)}>
                  {rate.name} ({rate.unit})
                </option>
              ))}
            </optgroup>
          </select>
        </div>
        <div>
          <button onClick={() => handleConvert()}>Refresh</button>
          {result && <p>{result}</p>}
        </div>
      </div>
    </div>
  );
};

export default Convertor;
